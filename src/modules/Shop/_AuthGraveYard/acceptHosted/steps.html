AuthorizeNetIFrame.onReceiveCommunication = function (querystr) { var params = parseQueryString(querystr); switch (params["action"]) { case "successfulSave": break; case "cancel": break; case "resizeWindow": var w = parseInt(params["width"]); var h = parseInt(params["height"]); var ifrm = document.getElementById("add_payment"); ifrm.style.width = w.toString() + "px"; ifrm.style.height = h.toString() + "px"; break; case "transactResponse": var ifrm = document.getElementById("add_payment"); ifrm.style.display = 'none'; } };
